y = data["Felicidad"]

# Calcular promedios
x_mean = np.mean(x)
y_mean = np.mean(y)

# Calcular coeficientes
B1 = np.sum((x - x_mean) * (y - y_mean)) / np.sum((x - x_mean) ** 2)
B0 = y_mean - B1 * x_mean

print("B1 =", B1)
print("B0 =", B0)

# Paso 6: Calcular error estándar y verificar significancia
n = len(y)
